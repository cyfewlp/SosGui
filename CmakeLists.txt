cmake_minimum_required(VERSION 4.2)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_INTERPROCEDURAL_OPTIMIZATION ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
option(BUILD_TESTING "Build tests" ON)

if (BUILD_TESTING)
  list(APPEND VCPKG_MANIFEST_FEATURES "test-deps")
endif ()

set(EXTERN_DIR ${CMAKE_CURRENT_SOURCE_DIR}/extern)
set(JAMIE_MODS_DIR ${EXTERN_DIR}/JamieMods)
set(CommonLibSSE_NG_DIR ${EXTERN_DIR}/CommonLibSSE-NG)
set(TEST_PROJ_NAME "${PROJECT_NAME}Test")

project(
    SosGui
    VERSION 0.2.0
    DESCRIPTION "SkyrimOutfitSystemSE ImGui Frontend"
    LANGUAGES CXX
)

configure_file(
    ${JAMIE_MODS_DIR}/cmake/version.rc.in
    ${CMAKE_CURRENT_BINARY_DIR}/version.rc
    @ONLY
)

## Set Output folder

if (DEFINED ENV{MO2_MODS_PATH} AND IS_DIRECTORY "$ENV{MO2_MODS_PATH}")
    set(OUTPUT_FOLDER "$ENV{MO2_MODS_PATH}/${PROJECT_NAME}")
endif ()

## Plugin setttings

find_package(spdlog CONFIG REQUIRED)
find_path(DETOURS_INCLUDE_DIRS "detours/detours.h")
find_library(DETOURS_LIBRARY detours REQUIRED)
find_package(Freetype REQUIRED) # for imgui_freetype

file(GLOB_RECURSE headers src/*.h)
file(GLOB_RECURSE sources src/*.cpp)

source_group(
    TREE ${CMAKE_CURRENT_SOURCE_DIR}
    FILES
    ${sources} ${headers}
)

include(${JAMIE_MODS_DIR}/cmake/imgui-config.cmake)
include(${EXTERN_DIR}/CommonLibSSE-NG/cmake/CommonLibSSE.cmake)
set(_OLD_BUILD_TESTS ${BUILD_TESTS})
set(BUILD_TESTS OFF) # forbid CommonLibSSE-NG from building its tests, since we only want to build our own tests, and CommonLibSSE-NG's tests are not relevant to us and would just slow down the build
add_subdirectory(${CommonLibSSE_NG_DIR} EXCLUDE_FROM_ALL)
set(BUILD_TESTS ${_OLD_BUILD_TESTS})

add_commonlibsse_plugin(${PROJECT_NAME} SOURCES
    ${sources}
    ${headers}
    ${ImGui_Basic_Sources}
    ${ImGui_Backend_Win32_Dx11_Sources}
    ${ImGui_FreeType_Sources}
    ${JAMIE_MODS_COMMON_DIR}/imgui/ImThemeLoader.cpp
    ${JAMIE_MODS_COMMON_DIR}/fonts/FontManager.cpp
    ${JAMIE_MODS_COMMON_DIR}/i18n/translator_manager.cpp
    ${IMGUIEX_DIR}/imguiex/ErrorNotifier.cpp
    ${IMGUIEX_DIR}/imguiex/imgui_manager.cpp
    ${EXTERN_DIR}/tracy/public/TracyClient.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/version.rc
    MINIMUM_SKSE_VERSION "1.5.97"
    AUTHOR "Jamie"
)
target_compile_features(${PROJECT_NAME} PRIVATE cxx_std_23)
target_precompile_headers(${PROJECT_NAME} PRIVATE src/PCH.h)
target_compile_definitions(${PROJECT_NAME} PRIVATE IMGUI_USER_CONFIG="${CMAKE_CURRENT_SOURCE_DIR}/imconfig.h")
if (CMAKE_BUILD_TYPE STREQUAL "Debug")
    target_compile_definitions(${PROJECT_NAME} PRIVATE TRACY_ENABLE)
endif ()

# include directories
target_include_directories(
    ${PROJECT_NAME}
    PRIVATE
    ${IMGUI_INCLUDE_DIRS}
    ${JAMIE_MODS_COMMON_DIR}
    ${JAMIE_MODS_EXTERN_DIR}
    ${IMGUIEX_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/src
    ${DETOURS_INCLUDE_DIRS}
    ${EXTERN_DIR}/tracy/public
)

# linkage
target_link_libraries(
    ${PROJECT_NAME}
    PRIVATE
    spdlog::spdlog
    CommonLibSSE
    ${DETOURS_LIBRARY}
    Freetype::Freetype
)
target_compile_options(${PROJECT_NAME} PRIVATE
    -Wpedantic -Wshadow -Wconversion -Wno-unused-command-line-argument
    -clang:-ftemplate-backtrace-limit=0
    -clang:-fno-elide-type
    -clang:-fdiagnostics-show-template-tree
)

include(${JAMIE_MODS_DIR}/cmake/build-lucide-icon.cmake)

add_build_lucide_icons(${PROJECT_NAME})

############################################################

if (BUILD_TESTING)
    include(${CMAKE_CURRENT_SOURCE_DIR}/test.cmake)
endif ()

# When your SKSE .dll is compiled, this will automatically copy the .dll into your mods folder.
# Only works if you configure DEPLOY_ROOT above (or set the SKYRIM_MODS_FOLDER environment variable)
if (DEFINED OUTPUT_FOLDER)
    # If you specify an <OUTPUT_FOLDER> (including via environment variables)
    # then we'll copy your mod files into Skyrim or a mod manager for you!

    # Copy the SKSE plugin .dll files into the SKSE/Plugins/ folder
    set(DLL_FOLDER "${OUTPUT_FOLDER}/SKSE/Plugins")
    set(TRANSLATION_FOLDER "${OUTPUT_FOLDER}/Interface/translations")
    set(Theme_FOLDER "${OUTPUT_FOLDER}/Interface/SosGui")

    message(STATUS "SKSE plugin output folder: ${DLL_FOLDER}")

    add_custom_command(
        TARGET "${PROJECT_NAME}"
        POST_BUILD
        COMMAND "${CMAKE_COMMAND}" -E make_directory "${DLL_FOLDER}"
        COMMAND "${CMAKE_COMMAND}" -E copy_if_different "$<TARGET_FILE:${PROJECT_NAME}>" "${DLL_FOLDER}/$<TARGET_FILE_NAME:${PROJECT_NAME}>"
        COMMAND "${CMAKE_COMMAND}" -E copy_if_different "${CMAKE_CURRENT_SOURCE_DIR}/contrib/Distribution/${PROJECT_NAME}_english.txt" "${TRANSLATION_FOLDER}/${PROJECT_NAME}_english.txt"
        COMMAND "${CMAKE_COMMAND}" -E copy_if_different "${CMAKE_SOURCE_DIR}/common/themes.toml.txt" "${Theme_FOLDER}/themes.toml"
        VERBATIM
    )

    # If you perform a "Debug" build, also copy .pdb file (for debug symbols)
    if (CMAKE_BUILD_TYPE STREQUAL "Debug")
        add_custom_command(
            TARGET "${PROJECT_NAME}"
            POST_BUILD
            COMMAND "${CMAKE_COMMAND}" -E copy_if_different "$<TARGET_PDB_FILE:${PROJECT_NAME}>" "${DLL_FOLDER}/$<TARGET_PDB_FILE_NAME:${PROJECT_NAME}>"
            VERBATIM
        )
    endif ()
endif ()
